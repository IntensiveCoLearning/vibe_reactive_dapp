## 一、项目介绍

### 1.1 什么是 Reactive Network？

Reactive Network（睿应层）是 Web3 首个去中心化执行层，让部署在同链或跨链的互不相关合约实现响应式交互。

简单来说：它是一个完全链上、事件驱动的「if-this-then-that（条件触发）」网络，用于去中心化自动化，并简化工作流。

### 1.2 核心特点

-   **事件驱动**：合约可以监听其他链上的事件并自动执行
    
-   **跨链自动化**：实现 Origin → Reactive → Destination 的自动化流程
    
-   **条件触发**：支持设置条件判断，满足条件才执行回调
    
-   **无需中心化服务**：完全去中心化，不依赖中心化 relayer

## 三、核心概念

### 3.1 Subscribe-Trigger-Callback 模型

```

┌─────────────┐     事件      ┌─────────────┐    条件判断    ┌─────────────┐

│    Origin   │ ──────────▶  │   Reactive  │ ──────────▶  │ Destination │

│   (源链)    │              │   (中间层)  │              │   (目标链)  │

└─────────────┘              └─────────────┘              └─────────────┘

  触发事件                       监听事件                     执行回调

                                判断条件

```

### 3.2 三段链路详解

| 环节 | 角色 | 说明 |

|------|------|------|

| Origin | 源合约 | 部署在源链上，当发生特定行为时 emit 事件 |

| Reactive | 中间层合约 | 部署在 Reactive Network 上，订阅 Origin 的事件，检测条件是否满足 |

| Destination | 目标合约 | 部署在目标链上，接收 Reactive 的回调，执行最终业务逻辑 |

### 3.3 适用场景

-   自动化止盈止损订单
    
-   清算保护
    
-   自动化投资组合再平衡
    
-   收益优化
    
-   跨链工作流
    
-   链上提醒通知
    
-   NFT 地板价监控

### 4.2 OriginContract.sol（源合约）

```solidity

// SPDX-License-Identifier: MIT

pragma solidity ^0.8.19;

  

contract OriginContract {

    event TriggerEvent(

        address indexed user,

        uint256 amount,

        string message

    );

  

    mapping(address => uint256) public lastAmount;

  

    function triggerAction(uint256 amount, string memory message) external {

        require(amount > 0, "Amount must be greater than 0");

        lastAmount[msg.sender] = amount;

        emit TriggerEvent(msg.sender, amount, message);

    }

}

```

### 4.3 ReactiveContract.sol（Reactive 中间层）

```solidity

// SPDX-License-Identifier: MIT

pragma solidity ^0.8.19;

  

import {Reactive} from "reactive/abstract/Reactive.sol";

import {IReactive} from "reactive/interfaces/IReactive.sol";

  

contract ReactiveContract is Reactive {

    event ReceivedEvent(

        uint256 srcChainId,

        address srcContract,

        address user,

        uint256 amount

    );

  

    uint256 public destinationChainId;

    address public destinationContract;

    uint256 public minAmount;

  

    constructor(

        uint256 _destinationChainId,

        address _destinationContract,

        uint256 _minAmount

    ) {

        destinationChainId = _destinationChainId;

        destinationContract = _destinationContract;

        minAmount = _minAmount;

    }

  

    function setMinAmount(uint256 _minAmount) external onlyOwner {

        minAmount = _minAmount;

    }

  

    function subscribe(

        uint256 chainId,

        address originContract,

        bytes32 eventSignature,

        uint256 interval

    ) external onlyOwner {

        react.subscribe(chainId, originContract, eventSignature, interval);

    }

  

    function react(

        uint256 srcChainId,

        address srcContract,

        bytes calldata data

    ) external override onlyReactor {

        (address user, uint256 amount, string memory message) = abi.decode(

            data,

            (address, uint256, string)

        );

  

        emit ReceivedEvent(srcChainId, srcContract, user, amount);

  

        if (amount >= minAmount) {

            react.callback(

                destinationChainId,

                destinationContract,

                abi.encode(user, amount, message)

            );

        }

    }

}

```

### 4.4 CallbackContract.sol（目标回调合约）

```solidity

// SPDX-License-Identifier: MIT

pragma solidity ^0.8.19;

  

import {IReactiveCallback} from "reactive/interfaces/IReactiveCallback.sol";

  

contract CallbackContract is IReactiveCallback {

    event CallbackReceived(

        address user,

        uint256 amount,

        string message,

        uint256 timestamp

    );

  

    mapping(address => uint256[]) public userAmounts;

    address[] public users;

  

    function receiveCrossChainCallback(

        uint256 srcChainId,

        address srcContract,

        bytes calldata data

    ) external override {

        (address user, uint256 amount, string memory message) = abi.decode(

            data,

            (address, uint256, string)

        );

  

        userAmounts[user].push(amount);

        if (userAmounts[user].length == 1) {

            users.push(user);

        }

  

        emit CallbackReceived(user, amount, message, block.timestamp);

    }

  

    function getUserHistory(address user) external view returns (uint256[] memory) {

        return userAmounts[user];

    }

}

```

### 4.5 部署脚本 Deploy.s.sol

```solidity

// SPDX-License-Identifier: MIT

pragma solidity ^0.8.19;

  

import "forge-std/Script.sol";

import {OriginContract} from "../src/OriginContract.sol";

import {ReactiveContract} from "../src/ReactiveContract.sol";

import {CallbackContract} from "../src/CallbackContract.sol";

  

contract DeployScript is Script {

    function run() external {

        uint256 deployerPrivateKey = vm.envUint("REACTIVE_LASNA_PRIVATE_KEY");

        vm.startBroadcast(deployerPrivateKey);

  

        // Chain IDs (Testnet)

        // Ethereum Sepolia: 11155111

        // Base Sepolia: 84532

        // Reactive Lasna: 5318007

  

        // 1. 部署 CallbackContract 到目标链（Base Sepolia）

        CallbackContract callback = new CallbackContract();

        console.log("CallbackContract deployed at:", address(callback));

  

        // 2. 部署 ReactiveContract 到 Reactive Lasna

        ReactiveContract reactive = new ReactiveContract(

            84532,                      // destinationChainId (Base Sepolia)

            address(callback),          // destinationContract

            100                         // minAmount

        );

        console.log("ReactiveContract deployed at:", address(reactive));

  

        // 3. 部署 OriginContract 到源链（Ethereum Sepolia）

        OriginContract origin = new OriginContract();

        console.log("OriginContract deployed at:", address(origin));

  

        // 4. 订阅事件

        reactive.subscribe(

            11155111,                  // sourceChainId (Ethereum Sepolia)

            address(origin),            // originContract

            bytes32(

                keccak256("TriggerEvent(address,uint256,string)")

            ),

            0                           // interval (0 = trigger on every event)

        );

        console.log("Subscribed to TriggerEvent");

  

        vm.stopBroadcast();

    }

}

```

## 十二、附录：常见问题

### Q1: 什么是 Reactive Contract？

A: Reactive Contract（响应式智能合约）是部署在 Reactive Network 上的智能合约，它可以监听其他链上的事件，并根据预设条件自动触发跨链回调。

### Q2: Reactive 和传统方案有什么区别？

A: 传统方案需要中心化的 bot 或服务轮询链上数据，而 Reactive 是完全去中心化的事件驱动架构，合约自动监听和执行。

### Q3: 哪些链可以作为 Origin/Destination？

A: 详见官方文档 Origins & Destinations 页面。主网支持 Ethereum、Arbitrum、Avalanche、Base、BSC 等；测试网支持 Ethereum Sepolia、Base Sepolia 等。

### Q4: 如何调试 Reactive 合约？

A: 使用 Reactscan ([https://reactscan.net](https://reactscan.net)) 查看链上交易和事件日志。
